Magnificent

A jQuery plugin for responsive zoom of images & more!

Version 2.x

Inner


<div mag-thumb="inner">
  <img src="img/alley/500x300.jpg" />
</div>
<div mag-zoom="inner">
  <img src="img/alley/1000x600.jpg" />
</div>
              

$host = $('[mag-thumb="inner"]');
$host.mag();

Outer


<div class="outer-wrap">
  <div mag-thumb="outer">
    <img src="img/alley/500x300.jpg" />
  </div>
  <div mag-zoom="outer">
    <img src="img/alley/1000x600.jpg" />
  </div>
</div>
              

$host = $('[mag-thumb="outer"]');
$host.mag({
  mode: 'outer'
});
